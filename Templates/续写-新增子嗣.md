<%*
// ── 续写家谱：新增子嗣模板 ──────────────────────────────────────
// 使用方法：在某位人物的笔记中打开命令面板，运行
//   "Templater: Open Insert Template Modal"，选择此模板即可

const parentFile = tp.config.active_file;

// 校验：必须在 People/ 目录的人物文件中运行
if (!parentFile || !parentFile.path.startsWith("People/")) {
  new Notice("⚠️ 请先打开 People/ 中的人物文件，再运行此模板", 5000);
  tR = "";
  return;
}

const parentName = parentFile.basename;

// ── 输入子嗣信息 ────────────────────────────────────────────────
const childName = await tp.system.prompt(`为【${parentName}】新增子嗣\n\n请输入子嗣姓名：`);
if (!childName) { tR = ""; return; }

const gender = await tp.system.suggester(
  ["男", "女"],
  ["男", "女"],
  false,
  "请选择性别"
);
if (!gender) { tR = ""; return; }

// ── 生成子嗣档案内容 ────────────────────────────────────────────
const childContent = `---
姓名: ${childName}
性别: ${gender}
父亲: "[[${parentName}]]"
母亲: 
配偶: []
子女: []
兄弟姐妹: []
备注: 
---

# ${childName}

## 家庭关系

### 父母
- 父：[[${parentName}]]
- 母：

### 配偶

### 子女

### 兄弟姐妹

## 备注

---

## 续写家谱

> [!tip] 为此人物新增子嗣
> 1. 打开命令面板 \`Ctrl/Cmd + P\`
> 2. 运行：\`Templater: Open Insert Template Modal\`
> 3. 选择模板：\`Templates/续写-新增子嗣\`
> 4. 按提示填写姓名和性别
>
> *系统将自动以当前人物为父亲，创建并打开新档案*
`;

// ── 在 People/ 中创建新文件并打开 ──────────────────────────────
await tp.file.create_new(childContent, childName, true, "People");

// ── 操作完成提示 ────────────────────────────────────────────────
new Notice(
  `✅ 已创建【${childName}】的档案\n\n请记得：\n① 在【${parentName}】的子女列表中添加 [[${childName}]]\n② 在家族关系图中添加对应节点`,
  8000
);

// 不在父文件中插入任何内容
tR = "";
_%>
